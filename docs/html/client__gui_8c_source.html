475int launch_gui_client_gui()
476{
477 printf("Launching GUI...\n");
478 int status;
479
480 // Create the client socket
481 if ((sock_client_gui = socket(AF_INET, SOCK_STREAM, 0)) < 0)
482 {
483 printf("\n Socket creation error \n");
484 return -1;
485 }
486
487 struct sockaddr_in serv_addr;
488 serv_addr.sin_family = AF_INET;
489 serv_addr.sin_port = htons(port_client_gui);
490
491 if (inet_pton(AF_INET, server_ip_client_gui, &serv_addr.sin_addr) <= 0)
492 {
493 printf("\nInvalid address/ Address not supported \n");
494 return -1;
495 }
496
497 if (connect(sock_client_gui, (struct sockaddr *)&serv_addr, sizeof(serv_addr)) < 0)
498 {
499 printf("\nConnection Failed \n");
500 return -1;
501 }
502
503 pthread_t recv_thread;
504 pthread_create(&recv_thread, NULL, start_client_receive_client_gui, NULL);
505 pthread_detach(recv_thread);

508 g_signal_connect(app_client_gui, "activate", G_CALLBACK(active_gui_client_gui), NULL);

510 g_object_unref(app_client_gui);
